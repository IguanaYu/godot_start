# 对话 + 剧情管理系统 — 详细设计方案（v2）

## Context

在现有 Richman 生存 Roguelike 游戏上搭建一套 RPG 级对话与剧情管理系统。
核心思路：**对话系统为核心，任务系统只是壳子，Iguana AI 负责全局剧情调度**。

用户明确的需求变更：
1. 任务系统先做壳子（只管完成/未完成 + 发信号），不写具体任务逻辑
2. 对话图之间也需要解锁管理（A 完成才能解锁 B）
3. 引入 "Iguana" 剧情管理 AI，根据玩家状态推内容
4. 引入地图探索进度系统
5. 对话 UI 需要新建

---

## 二、基础数据结构

### 2.1 对话图数据（DialogueGraph）

每个 DialogueGraph 是一段完整对话，存为 `.tres` Resource 文件。

```gdscript
# dialogue_graph.gd
class_name DialogueGraph
extends Resource

@export var graph_id: String = ""                # 唯一 ID，如 "elder_first_meet"
@export var npc_id: String = ""                  # 归属 NPC
@export var priority: int = 0                    # 同 NPC 多个可触发对话时的优先级
@export var repeatable: bool = false             # 是否可重复触发
@export var nodes: Array[DialogueNodeData] = []  # 节点列表
@export var connections: Array[DialogueConnection] = []  # 连接列表

# ── 前置条件 ──
@export var prerequisite_graph_ids: Array[String] = []   # 需要先完成的对话图
```

## 三、任务壳子系统（QuestShell）

### 3.1 设计思路

任务系统**只做最轻量的实现**：
- 每个任务有一个 ID 和一个状态（locked / active / completed）
- 任务可以被对话动作 `START_QUEST` 激活，被 `COMPLETE_QUEST` 完成
- 任务完成时发出信号 `quest_completed(quest_id)`
- 对话图可以引用任务状态作为前置条件
- **任务的具体内容（做什么、怎么判定完成）暂不实现**，以后再填

## 四、对话图管理层（DialogueManager）

### 4.1 职责

- 管理**所有对话图**的加载和缓存

- 对话图完成时，检查是否解锁了新的对话图

### 4.3 对话图解锁检查流程

```
某个事件发生（对话完成/任务完成/标记设置/天数变化/探索度变化）
  → DialogueManager.check_and_unlock_graphs()
    → 遍历所有 未解锁 的对话图
      → 检查每个前置条件组：
        ├── prerequisite_graph_ids: 全部在 _completed_graphs 中？
        ├── prerequisite_quest_ids: QuestManager 全部 completed？
        ├── prerequisite_min_day: GameManager.current_day_number >= ?
        ├── prerequisite_min_exploration: ExplorationProgress >= ?
        └── prerequisite_flags: _flags 中全部为 true？
      → 根据 prerequisite_logic（AND/OR）组合结果
      → 全部满足 → 解锁，发出 dialogue_graph_unlocked 信号
```

## 五、对话运行时（DialogueRunner）

### 5.1 职责

- 执行一个对话图，处理节点流转
- 驱动 DialogueUI 显示
- 触发节点上的动作

### 5.2 实现

```gdscript
# dialogue_runner.gd
class_name DialogueRunner
extends Node

signal dialogue_started(graph_id: String)
signal dialogue_ended(graph_id: String)
signal dialogue_action_triggered(action: ActionData)

var _current_graph: DialogueGraph = null
var _current_node: DialogueNodeData = null
var _is_active: bool = false

@export var dialogue_ui: PackedScene   # DialogueUI 场景引用
var _ui_instance: Control = null

func start_dialogue(graph: DialogueGraph) -> void:
    _current_graph = graph
    _current_node = _find_start_node(graph)
    _is_active = true
    _show_ui()
    dialogue_started.emit(graph.graph_id)
    _process_current_node()

func advance(choice_index: int = -1) -> void:
    # 根据当前节点类型推进：
    # - DIALOGUE: 找唯一连接的下一节点
    # - CHOICE: 根据 choice_index 找对应连接
    # - CONDITION: 评估条件，找 true/false 分支
    # - ACTION: 执行动作，自动前进到下一节点
    # - SUB_DIALOGUE: 保存当前状态，加载子对话
    # - END: 结束对话
    pass

func _process_current_node() -> void:
    match _current_node.node_type:
        NodeType.START:
            advance()  # 自动跳过开始节点
        NodeType.DIALOGUE:
            _ui_instance.show_dialogue(
                _current_node.speaker,
                _get_text(_current_node.text_key),
                _current_node.portrait_override
            )
            _execute_actions(_current_node.actions)
        NodeType.CHOICE:
            var visible_choices = _filter_visible_choices(_current_node.choices)
            _ui_instance.show_choices(visible_choices)
        NodeType.CONDITION:
            var result = _evaluate_condition(_current_node.condition)
            advance(result)  # true→port 0, false→port 1
        NodeType.ACTION:
            _execute_actions(_current_node.actions)
            advance()  # 动作节点自动前进
        NodeType.SUB_DIALOGUE:
            var sub_graph = DialogueManager.get_graph(_current_node.target_graph_id)
            # 保存现场，加载子对话
            pass
        NodeType.END:
            _end_dialogue()

func _end_dialogue() -> void:
    _is_active = false
    _hide_ui()
    DialogueManager.complete_graph(_current_graph.graph_id)
    dialogue_ended.emit(_current_graph.graph_id)

func _evaluate_condition(condition: ConditionData) -> bool:
    # 根据 condition_type 检查不同系统
    pass

func _execute_actions(actions: Array[ActionData]) -> void:
    for action in actions:
        match action.action_type:
            ActionType.SET_FLAG:

            ActionType.COMPLETE_QUEST:
```

### 6.2 DialogueUI 脚本

```gdscript
# dialogue_ui.gd
extends Control

signal advance_requested()
signal choice_made(index: int)

@export var typewriter_speed: float = 30.0  # 每秒字符数

@onready var dim_overlay: ColorRect = $DimOverlay
@onready var dialogue_box: PanelContainer = $DialogueBox
@onready var speaker_name: Label = $DialogueBox/SpeakerName
@onready var portrait: TextureRect = $DialogueBox/PortraitArea/Portrait
@onready var text_box: RichTextLabel = $DialogueBox/TextBox
@onready var indicator: TextureRect = $DialogueBox/Indicator
@onready var choice_container: VBoxContainer = $ChoiceContainer

var _is_typing: bool = false
var _full_text: String = ""
var _current_char: int = 0

func show_dialogue(speaker: String, text: String, portrait_texture: Texture2D = null):
    # 显示说话人名字、头像、文本（启动打字机效果）
    speaker_name.text = speaker
    if portrait_texture:
        portrait.texture = portrait_texture
        portrait.visible = true
    else:
        portrait.visible = false
    _start_typewriter(text)
    choice_container.visible = false
    dialogue_box.visible = true

func show_choices(choices: Array[ChoiceData]):
    # 显示选项按钮
    choice_container.visible = true
    for i in range(choices.size()):
        var button = choice_container.get_child(i)
        button.text = _get_text(choices[i].text_key)
        button.visible = true
    # 隐藏多余的按钮
    for i in range(choices.size(), choice_container.get_child_count()):
        choice_container.get_child(i).visible = false
    dialogue_box.visible = false

func _start_typewriter(text: String):
    _full_text = text
    _current_char = 0
    _is_typing = true
    text_box.text = ""
    _type_next_char()

func _type_next_char():
    if _current_char < _full_text.length():
        _current_char += 1
        text_box.text = _full_text.substr(0, _current_char)
        # 定时器回调 _type_next_char
    else:
        _is_typing = false
        indicator.visible = true

func _input(event: InputEvent):
    if not visible:
        return
    if event.is_action_pressed("ui_accept") or event.is_action_pressed("interact"):
        if _is_typing:
            # 跳过打字效果，直接显示全文
            _is_typing = false
            text_box.text = _full_text
            indicator.visible = true
        else:
            advance_requested.emit()
```

## 八、探索进度系统（ExplorationProgress）

### 8.1 设计思路

探索进度是一个 **0~1 的综合值**，由多个因素加权计算：

```gdscript
# exploration_progress.gd
class_name ExplorationProgress
extends Node

signal exploration_changed(new_value: float)
signal exploration_milestone_reached(milestone: String)

# 各维度的当前值
var quests_completed: int = 0
var total_coins_earned: int = 0
var dialogues_completed: int = 0
var areas_explored: int = 0        # 未来扩展
var enemies_defeated: int = 0       # 未来扩展

# 各维度的权重（可调整）
@export var quest_weight: float = 0.4
@export var coin_weight: float = 0.2
@export var dialogue_weight: float = 0.2
@export var area_weight: float = 0.1
@export var combat_weight: float = 0.1

# 归一化参考值（达到这个值算该维度满分）
@export var quest_max: int = 20
@export var coin_max: int = 1000
@export var dialogue_max: int = 30
@export var area_max: int = 10
@export var combat_max: int = 100

# 里程碑
@export var milestones: Dictionary = {
    0.1: "first_steps",
    0.25: "beginner_explorer",
    0.5: "seasoned_adventurer",
    0.75: "veteran_explorer",
    1.0: "master_explorer"
}

var _current_value: float = 0.0

func get_exploration_value() -> float:
    var quest_ratio = mini(float(quests_completed) / quest_max, 1.0)
    var coin_ratio = mini(float(total_coins_earned) / coin_max, 1.0)
    var dialogue_ratio = mini(float(dialogues_completed) / dialogue_max, 1.0)
    var area_ratio = mini(float(areas_explored) / area_max, 1.0)
    var combat_ratio = mini(float(enemies_defeated) / combat_max, 1.0)

    _current_value = (
        quest_ratio * quest_weight +
        coin_ratio * coin_weight +
        dialogue_ratio * dialogue_weight +
        area_ratio * area_weight +
        combat_ratio * combat_weight
    )
    return _current_value

func update_from_quest(quest_count: int):
    quests_completed = quest_count
    _check_milestones()
    exploration_changed.emit(get_exploration_value())

func update_from_coins(total_coins: int):
    total_coins_earned = total_coins
    _check_milestones()
    exploration_changed.emit(get_exploration_value())

func _check_milestones():
    var new_val = get_exploration_value()
    for threshold in milestones:
        if new_val >= threshold and _current_value < threshold:
            exploration_milestone_reached.emit(milestones[threshold])
            DialogueManager.set_flag("exploration_" + milestones[threshold])
```

## 九、Iguana 剧情管理 AI

### 9.1 设计思路

Iguana 是一个**内容调度器**，它的职责：
1. 监控玩家状态（探索进度、天数、已完成对话/任务）
2. 从已解锁的内容池中选择合适的内容推送给玩家
3. 决定地图上刷新什么情报/事件/据点
4. 决定是否在休息区生成新的 NPC

**核心原则**：Iguana 只能推送**已解锁**的内容。解锁逻辑在 DialogueManager（对话图解锁）和 QuestManager（任务解锁），Iguana 不管解锁，只管**推送**。

### 9.2 Iguana 的内容池

```gdscript
# iguana_content.gd
class_name IguanaContent
extends Resource

enum ContentType {
    INTEL,          # 情报（调查后刷新据点/收集物）
    EVENT,          # 地图事件（类似现有 SpecialEvent）
    NPC_SPAWN,      # NPC 出现在休息区
    QUEST_PUSH,     # 主动推送任务给玩家
    MAP_MODIFIER    # 地图修改（难度调整、刷新率等）
}

@export var content_id: String = ""
@export var content_type: ContentType
@export var display_name_key: String = ""
@export var description_key: String = ""

# 推送条件（Iguana 内部的条件，与解锁条件不同）
@export var min_exploration: float = 0.0
@export var min_day: int = 0
@export var max_day: int = 999         # 超过这天就不推了
@export var required_flags: Array[String] = []
@export var required_quests_completed: Array[String] = []
@export var required_graphs_completed: Array[String] = []
@export var required_content_completed: Array[String] = []  # 前置内容

# 推送权重（多个内容同时满足条件时，权重高的优先）
@export var push_weight: float = 1.0

# 是否已推送（运行时状态）
var pushed: bool = false
```

### 9.3 Iguana 管理器

```gdscript
# iguana.gd
class_name Iguana
extends Node

signal content_pushed(content: IguanaContent)
signal npc_should_spawn(npc_id: String, graph_id: String)

# 所有可用内容
var _content_pool: Array[IguanaContent] = []

# 已推送的内容
var _pushed_content: Array[String] = []

# 每次回到休息区时，Iguana 决定推送什么
func on_enter_rest_area():
    # 1. 收集所有满足推送条件的内容
    var candidates = _get_eligible_content()
    # 2. 按权重排序
    candidates.sort_custom(func(a, b): return a.push_weight > b.push_weight)
    # 3. 推送前 N 个（每次不要推太多）
    var to_push = candidates.slice(0, 2)  # 每次最多推 2 个新内容
    for content in to_push:
        _push_content(content)

func _get_eligible_content() -> Array[IguanaContent]:
    var result: Array[IguanaContent] = []
    for content in _content_pool:
        if content.pushed:
            continue
        if not _check_conditions(content):
            continue
        result.append(content)
    return result

func _check_conditions(content: IguanaContent) -> bool:
    var exp = ExplorationProgress.get_exploration_value()
    if exp < content.min_exploration:
        return false
    if GameManager.current_day_number < content.min_day:
        return false
    if GameManager.current_day_number > content.max_day:
        return false
    for flag in content.required_flags:
        if not DialogueManager.has_flag(flag):
            return false
    for qid in content.required_quests_completed:
        if not QuestManager.is_quest_completed(qid):
            return false
    for gid in content.required_graphs_completed:
        if not DialogueManager.is_graph_completed(gid):
            return false
    for cid in content.required_content_completed:
        if not _pushed_content.has(cid):
            return false
    return true

func _push_content(content: IguanaContent):
    content.pushed = true
    _pushed_content.append(content.content_id)
    content_pushed.emit(content)

    match content.content_type:
        ContentType.NPC_SPAWN:
            npc_should_spawn.emit(content.params.get("npc_id", ""),
                                   content.params.get("graph_id", ""))
        ContentType.INTEL:
            # 标记情报可用，下次进关卡时刷新在地图上
            DialogueManager.set_flag("intel_" + content.content_id)
        ContentType.QUEST_PUSH:
            QuestManager.start_quest(content.params.get("quest_id", ""))
        ContentType.EVENT:
            # 将事件加入当前关卡的 event_pool
            pass
        ContentType.MAP_MODIFIER:
            # 修改地图配置（难度、刷新率等）
            pass

# 存档
func get_save_data() -> Dictionary:
    return { pushed_content: _pushed_content }
```

### 9.4 Iguana 工作流程示意

```
玩家完成一个关卡 → 回到休息区
  → Iguana.on_enter_rest_area()
    → 收集所有"条件满足 + 未推送"的内容
    → 按权重排序，选择前 2 个推送
    → 例如：
      ├── 推送 INTEL "河流情报" → 设置标记 intel_river → 下个关卡地图上出现河流情报点
      └── 推送 NPC_SPAWN "神秘老人" → 休息区出现新 NPC

玩家下次进关卡 → 看到"河流情报"调查点
  → 调查 → 刷新河流区域据点 + 收集物
  → 收集完成 → COMPLETE_QUEST("explore_river")
  → QuestManager.quest_completed 信号
    → ExplorationProgress 更新
    → DialogueManager 检查解锁（"龙宫探索"对话图解锁了）
    → Iguana 检查推送条件

下次回到休息区 → Iguana.on_enter_rest_area()
  → 发现 "龙宫情报" 满足条件了（探索度达标 + river 任务完成）
  → 推送 INTEL "龙宫情报"
  → 同时可能推送新的 NPC 对话
```
